--<TABLEs>
SELECT
	TABLE_NAME AS "Name",
	CASE
		WHEN (TABLE_TYPE = 'VIEW') THEN
			CAST(1 AS unsigned)
		ELSE
			CAST(0 AS unsigned)
		END AS "isVT"
FROM INFORMATION_SCHEMA.TABLES
WHERE
	(
		(TABLE_TYPE = 'BASE TABLE')
		OR
		(TABLE_TYPE = 'VIEW')
	)
	AND
	(TABLE_SCHEMA != 'mysql')
ORDER BY "Name"
--</TABLEs>


--<COLUMNs>
SELECT
t1.TABLE_NAME,
	t1.COLUMN_NAME AS "Name",
	CASE
		WHEN t1.IS_NULLABLE = 'No' THEN
			CAST(0 AS unsigned)
		ELSE
			CAST(1 AS unsigned)
		END AS "isNullable",
	t1.DATA_TYPE AS "Type",
	t1.CHARACTER_MAXIMUM_LENGTH AS "Size",
	CASE
		WHEN ((t6.TABLE_TYPE != 'VIEW') AND (t1.COLUMN_KEY = 'PRI')) THEN
			CAST(1 AS unsigned)
		ELSE
			CAST(0 AS unsigned)
	END AS "isPK",
	CASE
		WHEN ((t6.TABLE_TYPE != 'VIEW') AND (t1.EXTRA = 'auto_increment')) THEN
			CAST(1 AS unsigned)
		ELSE
			CAST(0 AS unsigned)
	END AS "isIdentity",
	NULL AS "FK_TableName",
	NULL AS "FK_FieldName", 
	t1.COLUMN_DEFAULT, 
	t1.COLLATION_NAME
FROM INFORMATION_SCHEMA.COLUMNS t1
	LEFT JOIN INFORMATION_SCHEMA.TABLES t6 ON ((t6.TABLE_SCHEMA = t1.TABLE_SCHEMA) AND (t6.TABLE_NAME = t1.TABLE_NAME))
WHERE
	(t1.TABLE_SCHEMA = 'test')
	AND
	(
		(t1.TABLE_NAME = 'user')
		OR
		(t1.TABLE_NAME = 'UserGroup')
	)
ORDER BY t1.TABLE_NAME, t1.ORDINAL_POSITION
--</COLUMNs>
